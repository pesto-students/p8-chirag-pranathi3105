
1. Find the item that has minimum weight.

  SELECT min(item_weight) FROM react.items;

2. Find the different warehouses in “Pune”.

  SELECT w.wh_Name FROM react.cities c
  left join react.warehouse w on c.cityId = w.wh_city 
   where c.cityName = 'pune';

3. Find the details of items ordered by a customer “Mr. Patil”.

  SELECT * FROM react.order o
left join react.items i on o.order_item = i.item_id
left join react.customer c on o.order_cust = c.cust_id
 where c.cust_name  = 'Mr. Patil';

4. Find a Warehouse which has maximum stores.
    select * from react.warehouse where wh_City = (
SELECT st_city FROM (
SELECT COUNT(st_id)as count, st_City
FROM react.stores
GROUP BY st_City
HAVING COUNT(st_id) > 1) tab1
order by tab1.count desc
LIMIT 1)

5. Find an item which is ordered for a minimum number of times.

  SELECT order_item, count(order_id)
FROM react.order
GROUP BY order_item
ORDER BY count(order_id)  limit 1;

6. Find the detailed orders given by each customer.

  SELECT * FROM react.order o
left join react.items i on i.item_id = o.order_item
LEFT JOIN react.customer c ON c.cust_id = o.order_cust;
